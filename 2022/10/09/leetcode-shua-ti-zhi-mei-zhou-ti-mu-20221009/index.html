1、重新格式化电话号码

public String reformatNumber(String number) {
    // 1、格式化字段串，将类似 1-23-45 6 转为 123456 类型
    String s = number.replace(" ", "").replace("-", "");
    int n = s.length();
    StringBuilder sb = new StringBuilder();
    // 每隔三个组一组
    for (int i = 0; i < n; i += 3) {
        // 当拼接的字符串不是空，增加“-”
        if (sb.length() != 0) sb.append("-");
        
        if (i + 5 > n) {
            // 如果i+3已经超过n，表示剩余的字符在3以内，自成一个组
            if (i + 3 >= n) sb.append(s.substring(i));
            // 如果是4个，则分成两组拼接
            else sb.append(s.substring(i, i + 2)).append("-").append(s.substring(i + 2));
            break;
        }
        
        // 其他条件正常拼接
        sb.append(s.substring(i, i + 3));
    }
    return sb.toString();
}

2、在LR字符串中交换相邻字符串

按照题目说法，可以认为

  • 序号相同的 L : start 的下标不小于 end 的下标（即 L 不能往右移动）
  • 序号相同的 R : start 的下标不大于 end 的下标（即 R 不能往左移动）

可以根据这个逻辑去辨别可不可以去转

public boolean canTransform(String start, String end) {
    int n = start.length(), i = 0, j = 0;
    while (i < n || j < n) {
        // 在start字符串中，从左往右找到一个不是X的值
        while (i < n && start.charAt(i) == 'X') i++;
        // 在end字符串中，从左往右找到一个不是X的值
        while (j < n && end.charAt(j) == 'X') j++;
        /*
        	判断是否有一个串遍历完成，而另一个串遍历没有完成的情况,
        	如果出现了就只有两种结果：
            	1.两个串都只有X 分两串长度相同和长度不相同
            	2.其中一个串都是X，另一个串有一个其他字符
          	这两种情况都可以用i==j这个判断式来直接判断结果
        */
        if (i == n || j == n) return i == j;
        // 如果不相同则无法完成
        /*
        	例：
            	start: XXXXLXRXXXXX->XXXXXLRXXXXX  注:LR无法交换
            	end  : XXXXRXLXXXXX
        */
        if (start.charAt(i) != end.charAt(j)) return false;
        // 以start串的找到的值为例，如果是L,end串的L的索引则必须大于start串的L的索引
        if (start.charAt(i) == 'L' && i < j) return false;
        // 如果是R,end串的R的索引则必须小于start串的R的索引
        if (start.charAt(i) == 'R' && i > j) return false;
        // 其他情况可以完成交换，越过该次，进入下一次
        i++; j++;
    }
    return i == j;
}

3、检查二进制字符串字段

public boolean checkOnesSegment(String s) {
    int n = s.length(), cnt = 0, idx = 0;
    while (idx < n && cnt <= 1) {
        while (idx < n && s.charAt(idx) == '0') idx++;
        if (idx < n) {
            while (idx < n && s.charAt(idx) == '1') idx++;
            cnt++;
        }
    }
    return cnt <= 1;
}

4、使括号有效的最少添加

public int minAddToMakeValid(String s) {
    // res 记录插入次数
    int res = 0;
    // need 变量记录右括号的需求量
    int need = 0;

    for (int i = 0; i < s.length(); i++) {
        if (s.charAt(i) == '(') {
            // 对右括号的需求 + 1
            need++;
        }

        if (s.charAt(i) == ')') {
            // 对右括号的需求 - 1
            need--;

            if (need == -1) {
                need = 0;
                // 需插入一个左括号
                res++;
            }
        }
    }

    return res + need;
}

5、子域名访问计数

public List<String> subdomainVisits(String[] ss) {
    // 哈希表的运用
    Map<String, Integer> map = new HashMap<>();
    for (String s : ss) {
        int n = s.length(), idx = 0;
        while (idx < n && s.charAt(idx) != ' ') idx++;
        int cnt = Integer.parseInt(s.substring(0, idx));
        int start = idx + 1; idx = n - 1;
        while (idx >= start) {
            while (idx >= start && s.charAt(idx) != '.') idx--;
            String cur = s.substring(idx + 1);
            map.put(cur, map.getOrDefault(cur, 0) + cnt);
            idx--;
        }
    }
    return map.entrySet().stream().map(e -> e.getValue() + " " + e.getKey()).collect(Collectors.toList());
}

6、三等分

private int[] arr;

public int[] threeEqualParts(int[] arr) {
    int cnt = 0;
    int n = arr.length;
    // 1. 对1计数
    for (int v : arr) {
        cnt += v;
    }
    /*
    	2. 剪枝：
        	如果数量不是三的倍数，返回
            如果全是0，返回任意位置
    */
    if (cnt % 3 != 0) {
        return new int[]{-1, -1};
    }
    if (cnt == 0) {
        return new int[]{0, n - 1};
    }
    
    // 3. 将数组划分为每部分有count/3个1。三指针，分别指向第一个出现的1
    cnt /= 3;
    int i = find(1), j = find(cnt + 1), k = find(cnt * 2 + 1);
    
    // 开始向后遍历，逐个判断对应指针位置上的数字是否相等，如果发现不相等，返回[-1,-1]，否则继续往下走
    for (; k < n && arr[i] == arr[j] && arr[j] == arr[k]; ++i, ++j, ++k) {}
    
    // 如果最后k==n，返回[i-1, j]	
    return k == n ? new int[]{i - 1, j} : new int[]{-1, -1};
}

// 找到第x个1的位置
private int find(int x) {
    int s = 0;
    for (int i = 0; i < arr.length; ++i) {
        s += arr[i];
        if (s == x) {
            return i;
        }
    }
    return 0;
}

7、最大升序子数组和

public int maxAscendingSum(int[] nums) {
    int n = nums.length, ans = nums[0];
    for (int i = 1, cur = nums[0]; i < n; i++) {
        if (nums[i] > nums[i - 1]) cur += nums[i];
        else cur = nums[i];
        ans = Math.max(ans, cur);
    }
    return ans;
}

8、优势洗牌

public int[] advantageCount(int[] nums1, int[] nums2) {
    int n = nums1.length;
    Integer[] idx = new Integer[n]; // nums2索引
    // 赋予索引
    for (int i = 0; i < n; i++) idx[i] = i;
    // 对nums1排序
    Arrays.sort(nums1);
    Arrays.sort(idx, (i, j) -> nums2[i] - nums2[j]); //根据nums2数据升序，排序nums2(索引)
    int L = 0, R = n - 1; // nums2(索引)的左右指针
    for (int i : nums1) { // 遍历nums1
        if (i > nums2[idx[L]]) nums2[idx[L++]] = i; // 可以满足 nums1[i] > nums2[i]
        else nums2[idx[R--]] = i; //丢到数组最后
    }
    return nums2;
}

9、括号的分数

当遇到 ( 则存入一个占位数值 0，遇到 ) 取出栈顶元素 cur，根据栈顶元素数值分情况讨论：

  • 栈顶元素 cur = 0，即当前的 ) 的前一元素就是 ( ，根据 () 得一分的规则可知，我们本次操作得到的分值为 1；
  • 栈顶元素 cur 不为 0，即当前 ) 与其匹配的 ( 中间相隔了其他字符，根据 (A) 的得分规则，此时可知得分为 cur×2；

将两者结合可统一为 max(cur×2,1)

// 一般遇到括号题目，先想栈能否解决，本题利用栈来解决
public int scoreOfParentheses(String s) {
    Deque<Integer> d = new ArrayDeque<>();
    d.addLast(0);
    for (char c : s.toCharArray()) {
        if (c == '(') d.addLast(0);
        else {
            int cur = d.pollLast();
            d.addLast(d.pollLast() + Math.max(cur * 2 , 1));
        }
    }
    return d.peekLast();
}

另外看到另一个大佬的做法，用计数

public int scoreOfParentheses(String s) {
    int ans = 0, d = 0;
    for (int i = 0; i < s.length(); ++i) {
        if (s.charAt(i) == '(') {
            ++d;
        } else {
            --d;
            if (s.charAt(i - 1) == '(') {
                ans += 1 << d;
            }
        }
    }
    return ans;
}
